Needs["NETLink`"]
LoadNETAssembly["Tinkerforge",NotebookDirectory[]<>"../../.."]

host="localhost"
port=4223
uid="amb"(* Change to your UID *)

(* Create IPConnection and device object *)
ipcon=NETNew["Tinkerforge.IPConnection"]
dai=NETNew["Tinkerforge.BrickletIndustrialDualAnalogIn",uid,ipcon]
ipcon@Connect[host,port]

(* Set Period for (channel 1) for voltage callback to 1s (1000ms) *)
(* Note: The voltage callback is only called every second if the *)
(*       voltage has changed since the last call! *)
dai@SetVoltageCallbackPeriod[1,1000]

(* Callback function for current callback (parameter has unit mV) *)
VoltageCB[sender_,channel_,voltage_]:=
 Print["Voltage (channel "<>ToString[channel]<>"): "<>
       ToString[N[Quantity[voltage,"mV"]]]]
AddEventHandler[dai@Voltage,VoltageCB]

Input["Click OK to exit"]

(* Clean up *)
ipcon@Disconnect[]
ReleaseNETObject[dai]
ReleaseNETObject[ipcon]
